function createTarStream(filePaths) {

  const tempPaths = [];
 
  for (const originalPath of filePaths) {

    fs.copyFileSync(originalPath, tempTarget);
    tempPaths.push(tempTarget);
  }
 
  return tar.create({ cwd: tmpFolder }, tempPaths.map(p => path.basename(p)));
}
